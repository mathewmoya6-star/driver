-- ============================================ -- MEI DRIVE AFRICA - MODULE-BASED SCHEMA -- Each module has its own materials -- ============================================ -- Drop existing tables DROP TABLE IF EXISTS user_progress CASCADE; DROP TABLE IF EXISTS module_questions CASCADE; DROP TABLE IF EXISTS module_materials CASCADE; DROP TABLE IF EXISTS modules CASCADE; -- Table: modules (main categories) CREATE TABLE modules ( id SERIAL PRIMARY KEY, module_type TEXT UNIQUE NOT NULL CHECK (module_type IN ('learner', 'psv', 'ev', 'boda', 'schoolbus', 'academy')), name TEXT NOT NULL, icon TEXT, description TEXT, color TEXT, total_units INTEGER DEFAULT 0, is_active BOOLEAN DEFAULT TRUE, created_at TIMESTAMP DEFAULT NOW() ); -- Table: module_materials (units belong to specific module) CREATE TABLE module_materials ( id SERIAL PRIMARY KEY, module_id INTEGER REFERENCES modules(id) ON DELETE CASCADE, unit_number INTEGER NOT NULL, title TEXT NOT NULL, description TEXT, content TEXT, difficulty TEXT DEFAULT 'beginner', estimated_minutes INTEGER DEFAULT 45, display_order INTEGER DEFAULT 0, created_at TIMESTAMP DEFAULT NOW(), UNIQUE(module_id, unit_number) ); -- Table: module_questions (questions belong to specific module material) CREATE TABLE module_questions ( id SERIAL PRIMARY KEY, material_id INTEGER REFERENCES module_materials(id) ON DELETE CASCADE, question_number INTEGER NOT NULL, question_text TEXT NOT NULL, option_a TEXT NOT NULL, option_b TEXT NOT NULL, option_c TEXT NOT NULL, option_d TEXT NOT NULL, correct_answer TEXT CHECK (correct_answer IN ('A', 'B', 'C', 'D')), explanation TEXT, difficulty INTEGER DEFAULT 1, created_at TIMESTAMP DEFAULT NOW() ); -- Table: user_progress (tracks progress per material) CREATE TABLE user_progress ( id SERIAL PRIMARY KEY, user_id UUID NOT NULL, material_id INTEGER REFERENCES module_materials(id) ON DELETE CASCADE, completed BOOLEAN DEFAULT FALSE, quiz_score INTEGER, completed_at TIMESTAMP, last_accessed_at TIMESTAMP DEFAULT NOW(), UNIQUE(user_id, material_id) ); -- ============================================ -- INSERT MODULES -- ============================================ INSERT INTO
